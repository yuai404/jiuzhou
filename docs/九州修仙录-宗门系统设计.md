# 九州修仙录 - 宗门系统设计

> 版本：v0.1  
> 日期：2026-01-31  
> 目标：设计完整的宗门系统，支持宗门创建、成员管理、宗门建设、宗门任务、宗门战争与宗门福利。

## 1. 设计原则

- **社交核心**：宗门是玩家社交的主要载体，鼓励互助与协作
- **长期目标**：宗门建设提供长期成长线，增强归属感
- **公平竞争**：宗门战争有明确规则，避免大欺小
- **贡献激励**：个人贡献与宗门福利挂钩，避免白嫖
- **运营可控**：宗门规模、福利上限可配置

```sql
-- 宗门定义表
CREATE TABLE IF NOT EXISTS sect_def (
  id VARCHAR(64) PRIMARY KEY,
  name VARCHAR(64) UNIQUE NOT NULL,
  leader_id INT NOT NULL,
  level INT DEFAULT 1,
  exp BIGINT DEFAULT 0,
  funds BIGINT DEFAULT 0,
  reputation BIGINT DEFAULT 0,
  build_points INT DEFAULT 0,
  announcement TEXT,
  description TEXT,
  icon VARCHAR(256),
  join_type VARCHAR(32) DEFAULT 'apply',
  join_min_realm VARCHAR(64) DEFAULT '凡人',
  member_count INT DEFAULT 1,
  max_members INT DEFAULT 20,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_sect_name ON sect_def(name);
CREATE INDEX idx_sect_leader ON sect_def(leader_id);

-- 宗门成员表
CREATE TABLE IF NOT EXISTS sect_member (
  id BIGSERIAL PRIMARY KEY,
  sect_id VARCHAR(64) NOT NULL REFERENCES sect_def(id) ON DELETE CASCADE,
  character_id INT NOT NULL UNIQUE,
  position VARCHAR(32) NOT NULL DEFAULT 'disciple',
  contribution BIGINT DEFAULT 0,
  weekly_contribution INT DEFAULT 0,
  joined_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_sect_member_sect ON sect_member(sect_id);
CREATE INDEX idx_sect_member_char ON sect_member(character_id);

-- 宗门申请表
CREATE TABLE IF NOT EXISTS sect_application (
  id BIGSERIAL PRIMARY KEY,
  sect_id VARCHAR(64) NOT NULL REFERENCES sect_def(id) ON DELETE CASCADE,
  character_id INT NOT NULL,
  message TEXT,
  status VARCHAR(16) DEFAULT 'pending',
  created_at TIMESTAMPTZ DEFAULT NOW(),
  handled_at TIMESTAMPTZ,
  handled_by INT
);

-- 宗门建筑表
CREATE TABLE IF NOT EXISTS sect_building (
  id BIGSERIAL PRIMARY KEY,
  sect_id VARCHAR(64) NOT NULL REFERENCES sect_def(id) ON DELETE CASCADE,
  building_type VARCHAR(64) NOT NULL,
  level INT DEFAULT 1,
  status VARCHAR(32) DEFAULT 'normal',
  upgrade_start_at TIMESTAMPTZ,
  upgrade_end_at TIMESTAMPTZ,
  UNIQUE(sect_id, building_type)
);

-- 宗门任务进度表
CREATE TABLE IF NOT EXISTS sect_quest_progress (
  id BIGSERIAL PRIMARY KEY,
  character_id INT NOT NULL,
  quest_id VARCHAR(64) NOT NULL,
  progress INT DEFAULT 0,
  status VARCHAR(32) DEFAULT 'in_progress',
  accepted_at TIMESTAMPTZ DEFAULT NOW(),
  completed_at TIMESTAMPTZ,
  UNIQUE(character_id, quest_id)
);

-- 宗门日志表
CREATE TABLE IF NOT EXISTS sect_log (
  id BIGSERIAL PRIMARY KEY,
  sect_id VARCHAR(64) NOT NULL REFERENCES sect_def(id) ON DELETE CASCADE,
  log_type VARCHAR(32) NOT NULL,
  operator_id INT,
  target_id INT,
  content TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```

### 10.4 宗门福利计算

```typescript
// 宗门福利加成计算
interface SectBonuses {
  attrBonus: Record<string, number>;  // 属性加成
  expBonus: number;                    // 经验加成百分比
  dropBonus: number;                   // 掉落加成百分比
  craftBonus: number;                  // 制作加成百分比
}

function calculateSectBonuses(
  sectLevel: number,
  buildings: SectBuilding[],
  memberPosition: string
): SectBonuses {
  const bonuses: SectBonuses = {
    attrBonus: {},
    expBonus: 0,
    dropBonus: 0,
    craftBonus: 0,
  };
  
  // 宗门等级基础加成
  bonuses.expBonus += sectLevel * 2;  // 每级+2%经验
  
  // 建筑加成
  for (const building of buildings) {
    switch (building.building_type) {
      case 'library':  // 藏经阁
        bonuses.expBonus += building.level * 1;
        break;
      case 'training_hall':  // 练功房
        bonuses.attrBonus['wugong'] = (bonuses.attrBonus['wugong'] || 0) + building.level * 10;
        bonuses.attrBonus['fagong'] = (bonuses.attrBonus['fagong'] || 0) + building.level * 10;
        break;
      case 'alchemy_room':  // 丹房
        bonuses.craftBonus += building.level * 2;
        break;
      case 'spirit_array':  // 聚灵阵
        bonuses.attrBonus['lingqi_huifu'] = (bonuses.attrBonus['lingqi_huifu'] || 0) + building.level * 5;
        break;
    }
  }
  
  // 职位加成
  const positionBonus: Record<string, number> = {
    'leader': 20,
    'vice_leader': 15,
    'elder': 10,
    'elite': 5,
    'disciple': 0,
  };
  bonuses.expBonus += positionBonus[memberPosition] || 0;
  
  return bonuses;
}
```
