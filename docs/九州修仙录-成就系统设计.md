# 九州修仙录 - 成就系统设计

> 版本：v0.1  
> 日期：2026-02-01  
> 目标：设计完整的成就系统，支持多类型成就、进度追踪、成就点数、称号奖励与成就展示。

## 1. 设计原则

- **引导探索**：成就引导玩家体验各类玩法，发现隐藏内容
- **长期目标**：提供长期追求，增强游戏粘性
- **荣誉展示**：稀有成就是实力与投入的象征
- **奖励适度**：成就奖励以荣誉为主，实用奖励为辅
- **进度可视**：清晰展示进度，激励完成

## 2. 成就分类

### 2.1 按类型分类

| 类型 | 说明 | 示例 |
|------|------|------|
| 战斗成就 | 战斗相关 | 击杀怪物、Boss、PVP |
| 修炼成就 | 境界突破 | 达到特定境界 |
| 功法成就 | 功法修炼 | 功法等级、数量 |
| 装备成就 | 装备收集 | 品质、套装、强化 |
| 探索成就 | 地图探索 | 发现地点、隐藏区域 |
| 副本成就 | 秘境挑战 | 通关、评分、首杀 |
| 社交成就 | 社交互动 | 组队、宗门、好友 |
| 收集成就 | 物品收集 | 图鉴、稀有物品 |
| 生活成就 | 生活技能 | 炼丹、炼器、采集 |
| 特殊成就 | 隐藏/限时 | 彩蛋、活动 |

### 2.2 按稀有度分类

| 稀有度 | 颜色 | 成就点 | 说明 |
|--------|------|--------|------|
| 普通 | 白色 | 5 | 基础成就，易完成 |
| 精良 | 绿色 | 10 | 需要一定投入 |
| 稀有 | 蓝色 | 20 | 需要较多投入 |
| 史诗 | 紫色 | 50 | 高难度/长期目标 |
| 传说 | 橙色 | 100 | 极难/隐藏成就 |

## 4. 成就追踪机制

### 4.1 追踪类型

| 类型 | 说明 | 示例 |
|------|------|------|
| counter | 计数器，累计到目标值 | 击杀100只怪物 |
| flag | 标记，完成即达成 | 首次击杀Boss |
| multi | 多目标，完成所有子项 | 收集5种火系材料 |

### 4.3 事件触发流程

```
1. 游戏事件发生（击杀、通关、获得物品等）
2. 发送成就事件：{ trackKey, value, metadata }
3. 成就服务匹配相关成就
4. 更新进度
5. 检查是否完成
6. 完成则发送通知
```

```
POST /api/achievement/claim
Body: { achievement_id }
Response: { 
  success: boolean, 
  rewards: RewardList,
  title?: TitleInfo
}
```

```
GET /api/title/list
Response: { titles: TitleInfo[], equipped: string }

```

Response: { success: boolean }

### 11.4 事件处理器

```typescript
// server/src/events/achievementEventHandler.ts
import { gameEvents } from './GameEventEmitter.js';
import { GameEventType } from './GameEventTypes.js';
import { updateAchievementProgress } from '../services/achievementService.js';

export function registerAchievementEventHandlers() {
  // 战斗成就
  gameEvents.on(GameEventType.MONSTER_KILLED, async (event) => {
    // 击杀特定怪物
    await updateAchievementProgress(
      event.characterId,
      `kill:monster:${event.data.monsterId}`,
      1
    );
    // 击杀任意怪物
    await updateAchievementProgress(event.characterId, 'kill:monster:*', 1);
  });
  
  gameEvents.on(GameEventType.BOSS_KILLED, async (event) => {
    await updateAchievementProgress(
      event.characterId,
      `kill:boss:${event.data.bossId}`,
      1
    );
    await updateAchievementProgress(event.characterId, 'kill:boss:*', 1);
  });
  
  // 副本成就
  gameEvents.on(GameEventType.DUNGEON_CLEARED, async (event) => {
    await updateAchievementProgress(
      event.characterId,
      `dungeon:clear:${event.data.dungeonId}`,
      1
    );
    if (event.data.score === 'S') {
      await updateAchievementProgress(event.characterId, 'dungeon:score_s:*', 1);
    }
    if (event.data.isFirstClear) {
      await updateAchievementProgress(
        event.characterId,
        `dungeon:first_clear:${event.data.dungeonId}`,
        1
      );
    }
  });
  
  // 修炼成就
  gameEvents.on(GameEventType.REALM_BREAKTHROUGH, async (event) => {
    await updateAchievementProgress(
      event.characterId,
      `realm:reach:${event.data.newRealm}`,
      1
    );
  });
  
  gameEvents.on(GameEventType.TECHNIQUE_UPGRADED, async (event) => {
    await updateAchievementProgress(
      event.characterId,
      `skill:level:${event.data.techniqueId}:${event.data.newLevel}`,
      1
    );
  });
  
  // 探索成就
  gameEvents.on(GameEventType.MAP_DISCOVERED, async (event) => {
    await updateAchievementProgress(
      event.characterId,
      `map:discover:${event.data.mapId}`,
      1,
      { mapId: event.data.mapId }
    );
  });
  
  // 社交成就
  gameEvents.on(GameEventType.TEAM_CREATED, async (event) => {
    await updateAchievementProgress(event.characterId, 'team:create', 1);
  });
  
  gameEvents.on(GameEventType.SECT_JOINED, async (event) => {
    await updateAchievementProgress(event.characterId, 'sect:join', 1);
  });
  
  gameEvents.on(GameEventType.FRIEND_ADDED, async (event) => {
    await updateAchievementProgress(event.characterId, 'friend:add', 1);
  });
  
  // 收集成就
  gameEvents.on(GameEventType.ITEM_OBTAINED, async (event) => {
    await updateAchievementProgress(
      event.characterId,
      `item:obtain:${event.data.itemDefId}`,
      event.data.quantity
    );
    if (event.data.quality === '天') {
      await updateAchievementProgress(event.characterId, 'equip:quality:legendary', 1);
    }
  });
}
```

): Promise<void> {
  // 1. 查找匹配的成就定义
  const achievements = await findMatchingAchievements(trackKey);
  
  for (const achievement of achievements) {
    // 2. 获取或创建角色成就记录
    let charAch = await getOrCreateCharacterAchievement(characterId, achievement.id);
    
    if (charAch.status === 'completed' || charAch.status === 'claimed') {
      continue; // 已完成的跳过
    }
    
    // 3. 根据追踪类型更新进度
    switch (achievement.track_type) {
      case 'counter':
        charAch.progress += increment;
        break;
        
      case 'flag':
        charAch.progress = 1;
        break;
        
      case 'multi':
        // 多目标类型，记录已完成的子项
        const progressData = charAch.progress_data || [];
        if (metadata?.targetId && !progressData.includes(metadata.targetId)) {
          progressData.push(metadata.targetId);
          charAch.progress_data = progressData;
          charAch.progress = progressData.length;
        }
        break;
    }
    
    // 4. 检查是否完成
    if (charAch.progress >= achievement.target_value) {
      charAch.status = 'completed';
      charAch.completed_at = new Date();
      
      // 5. 发送完成通知
      await sendAchievementNotification(characterId, achievement);
    }
    
    // 6. 保存进度
    await saveCharacterAchievement(charAch);
  }
}

// 匹配成就定义（支持通配符）
async function findMatchingAchievements(trackKey: string): Promise<AchievementDef[]> {
  // 精确匹配
  const exact = await query(
    'SELECT * FROM achievement_def WHERE track_key = $1 AND enabled = true',
    [trackKey]
  );
  
  // 通配符匹配（如 kill:monster:* 匹配 kill:monster:wolf）
  const parts = trackKey.split(':');
  const wildcardKeys = [];
  for (let i = parts.length - 1; i >= 0; i--) {
    wildcardKeys.push([...parts.slice(0, i), '*'].join(':'));
  }
  
  const wildcard = await query(
    'SELECT * FROM achievement_def WHERE track_key = ANY($1) AND enabled = true',
    [wildcardKeys]
  );
  
  return [...exact.rows, ...wildcard.rows];
}
