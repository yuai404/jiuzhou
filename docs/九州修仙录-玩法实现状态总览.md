## 4. 全局事件系统（基础设施）

为了支持成就、任务、活动等系统的进度追踪，需要先实现全局事件系统。

### 4.1 事件类型定义

```typescript
// server/src/events/GameEventTypes.ts
export enum GameEventType {
  // 战斗事件
  MONSTER_KILLED = 'monster:killed',
  BOSS_KILLED = 'boss:killed',
  BATTLE_WON = 'battle:won',
  BATTLE_LOST = 'battle:lost',
  
  // 副本事件
  DUNGEON_ENTERED = 'dungeon:entered',
  DUNGEON_CLEARED = 'dungeon:cleared',
  DUNGEON_SCORE = 'dungeon:score',
  
  // 角色事件
  REALM_BREAKTHROUGH = 'realm:breakthrough',
  LEVEL_UP = 'level:up',
  
  // 功法事件
  TECHNIQUE_LEARNED = 'technique:learned',
  TECHNIQUE_UPGRADED = 'technique:upgraded',
  SKILL_USED = 'skill:used',
  
  // 物品事件
  ITEM_OBTAINED = 'item:obtained',
  ITEM_USED = 'item:used',
  EQUIPMENT_EQUIPPED = 'equipment:equipped',
  
  // 社交事件
  TEAM_CREATED = 'team:created',
  TEAM_JOINED = 'team:joined',
  FRIEND_ADDED = 'friend:added',
  SECT_JOINED = 'sect:joined',
  
  // 交易事件
  ITEM_SOLD = 'item:sold',
  ITEM_BOUGHT = 'item:bought',
  
  // 探索事件
  MAP_DISCOVERED = 'map:discovered',
  NPC_TALKED = 'npc:talked',
  
  // 制作事件
  ITEM_CRAFTED = 'item:crafted',
  RECIPE_LEARNED = 'recipe:learned',
}

export interface GameEvent {
  type: GameEventType;
  characterId: number;
  userId: number;
  data: Record<string, any>;
  timestamp: number;
}
```

### 4.2 事件发射器

```typescript
// server/src/events/GameEventEmitter.ts
import { EventEmitter } from 'events';
import { GameEvent, GameEventType } from './GameEventTypes.js';

class GameEventEmitter extends EventEmitter {
  private static instance: GameEventEmitter;
  
  static getInstance(): GameEventEmitter {
    if (!this.instance) {
      this.instance = new GameEventEmitter();
    }
    return this.instance;
  }
  
  emit(event: GameEventType, data: Omit<GameEvent, 'type' | 'timestamp'>): boolean {
    const fullEvent: GameEvent = {
      type: event,
      ...data,
      timestamp: Date.now(),
    };
    return super.emit(event, fullEvent);
  }
  
  // 便捷方法
  emitMonsterKilled(characterId: number, userId: number, monsterId: string, monsterName: string) {
    this.emit(GameEventType.MONSTER_KILLED, {
      characterId,
      userId,
      data: { monsterId, monsterName },
    });
  }
  
  emitDungeonCleared(characterId: number, userId: number, dungeonId: string, score: string) {
    this.emit(GameEventType.DUNGEON_CLEARED, {
      characterId,
      userId,
      data: { dungeonId, score },
    });
  }
  
  // ... 其他便捷方法
}

export const gameEvents = GameEventEmitter.getInstance();
```

### 4.3 事件监听器注册

```typescript
// server/src/events/registerEventListeners.ts
import { gameEvents } from './GameEventEmitter.js';
import { GameEventType } from './GameEventTypes.js';
import { updateQuestProgress } from '../services/questService.js';
import { updateAchievementProgress } from '../services/achievementService.js';
import { updateEventProgress } from '../services/eventService.js';

export function registerEventListeners() {
  // 任务系统监听
  gameEvents.on(GameEventType.MONSTER_KILLED, async (event) => {
    await updateQuestProgress(event.characterId, 'kill', event.data.monsterId);
  });
  
  gameEvents.on(GameEventType.DUNGEON_CLEARED, async (event) => {
    await updateQuestProgress(event.characterId, 'dungeon', event.data.dungeonId);
  });
  
  // 成就系统监听
  gameEvents.on(GameEventType.MONSTER_KILLED, async (event) => {
    await updateAchievementProgress(event.characterId, `kill:monster:${event.data.monsterId}`, 1);
    await updateAchievementProgress(event.characterId, 'kill:monster:*', 1);
  });
  
  // 活动系统监听
  gameEvents.on(GameEventType.DUNGEON_CLEARED, async (event) => {
    await updateEventProgress(event.characterId, 'dungeon_clear', event.data);
  });
  
  console.log('✓ 游戏事件监听器已注册');
}
```

## 7. 技术注意事项

### 7.1 数据库事务
- 涉及多表操作的功能必须使用事务
- 特别是：任务奖励发放、副本结算、交易

### 7.2 并发控制
- 组队副本需要处理多人同时操作
- 拍卖行竞拍需要乐观锁/悲观锁
- 宗门资源操作需要原子性

### 7.3 缓存策略
- 任务定义、副本定义等静态数据可缓存
- 排行榜数据定时刷新
- 在线状态使用Redis

### 7.4 实时通信
- 组队副本战斗状态同步
- 聊天消息推送
- 好友在线状态
